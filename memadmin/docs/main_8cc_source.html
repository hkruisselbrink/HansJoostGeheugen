94  char options[] = "s:a:tvcrfFnN"; // De opties die we willen herkennen
95  //

99  // Als je alle algoritmes zou realiseren dan wordt
100  // het iets in de trant van: "s:a:tvcrfFnNbBwWpm2"
101  //
102  // Algemene opties:
103  // "s:" staat voor: -s xxx = omvang van het beheerde geheugen (in "eenheden")
104  // "a:" staat voor: -a xxx = aantal alloc/free acties (aka new/delete)
105  // "t" staat voor: -t = code testen (i.p.v. performance meten)
106  // "v" staat voor: -v = verbose mode (vertel wat er gebeurt)
107  // "c" staat voor: -c = check mode (bewaak 'free' acties)
108  //
109  // Opties om een beheeralgoritme uit te kiezen ...
110  // r staat voor: -r = random-fit allocator
111  // f staat voor: -f = first-fit allocator (lazy)
112  // F staat voor: -F = first-fit allocator (eager)
113  // n staat voor; -n = next-fit allocator (lazy)
114  // N staat voor; -N = next-fit allocator (eager)
115  // b staat voor; -b = best-fit allocator (lazy)
116  // B staat voor; -b = best-fit allocator (eager)
117  // w staat voor; -w = worst-fit allocator (lazy)
118  // W staat voor; -w = worst-fit allocator (eager)
119  // enz
120  // 2 staat voor: -2 = buddy allocator
121  //
122  // Voor meer informatie, zie: man 3 getopt
123  //
124 
125  int opt = 0; // Het "huidige" optie karakter (of -1).
126 
127  do {
128  // Haal een optie uit argc/argv
129  // (en zet zonodig het bijbehorende argument in 'optarg')
130  opt = getopt(argc, argv, options);
131  switch (opt) { // welke optie is dit?
132  // ALGEMEEN
133  case 's': // the size of the (imaginary) memory being managed
134  size = atol(optarg); // atol = "ascii to long"
135  break;
